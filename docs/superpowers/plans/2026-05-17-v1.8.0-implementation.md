## Pattern X reminder — how to write each C# block

For every section closure, the recipe is:

1. Find the section in its file (paths listed per task).
2. Read the existing GDScript code block to understand intent.
3. **Immediately after** the closing ``` ``` ``` of the `gdscript` block, append a blank line then a `csharp` fenced block that translates the same example to idiomatic Godot 4 C#.
4. Do **not** rewrite the GDScript or change prose.
5. Apply Godot C# conventions (full reference: `skills/csharp-godot/SKILL.md`):
   - PascalCase methods matching Godot API (`_Ready`, `_Process`, `_PhysicsProcess`, `_UnhandledInput`)
   - `GetNode<T>("path")` instead of `get_node("path")`
   - `[Export] public T MyField` instead of `@export var my_field: T`
   - `[Signal] public delegate void FooEventHandler(args)` instead of `signal foo(args)`
   - `EmitSignal(SignalName.Foo, args)` instead of `emit_signal("foo", args)`
   - `public partial class Foo : Node` (always `partial`, always inherit explicitly)
   - `await ToSignal(obj, Signal.SignalName.Bar)` instead of `await obj.bar`
   - `Callable.From(MyMethod)` instead of `Callable(self, "my_method")`
   - Use `Godot.Collections.Dictionary` / `Godot.Collections.Array<T>` for Variant-typed containers exposed to the editor; use `System.Collections.Generic.Dictionary` for internal C#-only state
6. Per-section C# is short to medium (5–40 lines). Don't over-engineer — match the GDScript example's scope.

    public override void _PhysicsProcess(double delta)

using Godot;

public partial class WallJumpPlayer : CharacterBody2D
{
    [Export] public float WallJumpForce = 400f;
    [Export] public float WallSlideSpeed = 60f;
    [Export] public float Gravity = 980f;

    public override void _PhysicsProcess(double delta)
    {
        var velocity = Velocity;

        if (IsOnWall() && !IsOnFloor())
        {
            velocity.Y = Mathf.Min(velocity.Y + Gravity * (float)delta, WallSlideSpeed);

            if (Input.IsActionJustPressed("jump"))

                velocity = wallNormal * WallJumpForce + Vector2.Up * WallJumpForce;
            }
        }

            velocity.Y += Gravity * (float)delta;
        }

        Velocity = velocity;
        MoveAndSlide();
    }
}

using Godot;

// State data lives on a Resource subclass
public partial class StateResource : Resource
{
    [Export] public string StateName;
    [Export] public Godot.Collections.Dictionary<string, string> Transitions = new();
    [Export] public string EnterMethod;
    [Export] public string UpdateMethod;
    [Export] public string ExitMethod;
}

public partial class DataDrivenStateMachine : Node
{
    [Export] public Godot.Collections.Array<StateResource> States = new();
    [Export] public string InitialState;

    private StateResource _current;
    private GodotObject _owner;

    public override void _Ready()
    {
        _owner = GetParent();
        TransitionTo(InitialState);
    }

    public override void _PhysicsProcess(double delta)
    {
        if (!string.IsNullOrEmpty(_current?.UpdateMethod))
            _owner.Call(_current.UpdateMethod, delta);
    }

    public void Send(string trigger)
    {
        if (_current.Transitions.TryGetValue(trigger, out var target))
            TransitionTo(target);
    }

    private void TransitionTo(string stateName)
    {
        if (_current != null && !string.IsNullOrEmpty(_current.ExitMethod))
            _owner.Call(_current.ExitMethod);

        foreach (var s in States)
        {
            if (s.StateName == stateName)
            {
                _current = s;
                break;
            }
        }

        if (!string.IsNullOrEmpty(_current?.EnterMethod))
            _owner.Call(_current.EnterMethod);
    }
}
